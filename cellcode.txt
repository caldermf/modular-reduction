set G = SO(2,3).complexification
set p = G.trivial
set (param_list,cells) = block_and_cells(p)
set w_list = for p in param_list do p.x.w.word[:length(p.x)] od
set cell_lists = for C in cells do C.labels od
set cell_word_lists = for list in cell_lists do for m in list do w_list[m] od od
set lcells = for i:#cell_lists do (G,cells[i]).left_cells od
set list_of_lcell_lists = for i:#cell_lists do for C in lcells[i] do C.labels od od
set list_of_lcell_word_lists = for i:#cell_lists do for list in list_of_lcell_lists[i] do for m in list do w_list[m] od od od
list_of_lcell_word_lists
